ospf.yml
---
-   name: Configure OSPF
    hosts: routers
    gather_facts: false
    connection: local

    vars:
        cli:
            username: admin
            password: admin
    
    tasks:
        -   name: ospf
            ios_config:
                provider:"{{ cli }}
                authorize: yes
                parents: router ospf 1
                lines:
                    - network 0.0.0.0 255.255.255.255 area 0 #c'est un test
                register: print_output

        -   debug: var=print_output
